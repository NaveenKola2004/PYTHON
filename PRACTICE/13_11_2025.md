```python
def Sum(n):
    sum=0
    while n>0:
        digit=n%10
        sum+=digit
        n//=10
    return sum
def Max_digit_sum(start,stop):
    l=[]
    for i in range(start,stop+1):
        go=Sum(i)
        l.append(go)
    n=len(l)
    for i in range(n-1):
        for j in range(n-i-1):
            if (l[j]<l[j+1]):
                temp=l[j]
                l[j]=l[j+1]
                l[j+1]=temp
    return l[0]
print(Max_digit_sum(50,75))
```
