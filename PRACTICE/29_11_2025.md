def Sum(n):
    sum=0
    while n>0:
        digit=n%10
        sum+=digit**2
        n//=10
    return sum
def Solution(num):
    seen=set()
    store=num
    while store>1:
        if store in seen:
            return False
        seen.add(store)
        store=Sum(store)
    return True
