```python
def Solution(a):
    word=""
    i=0
    while i<len(a):
        ch=a[i]
        i+=1
        num=""
        while i<len(a) and a[i].isdigit():
            num+=a[i]
            i+=1
        word+=ch*int(num)
    return word
a=input()
print(Solution(a))
```

```python
    b=[]
    for i in range(1,k+1):
        word=a[-i:]+a[:-i]
        b.append(word)
    return b[k-1]
```
