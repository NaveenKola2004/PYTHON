# 1 problem Magic number 

```python

Find the sum of digits

If the sum has more than one digit, repeat the process

If the final result is 1, it's a Magic number

```

def count(num):
    sum_num=0
    count=0
    while num>0:
        digit=num%10
        sum_num+=digit
        count+=1
        num//=10
    return sum_num,count
def Solution(n):
    return_value=list(count(n))[0]
    return_count=list(count(n))[1]
    while return_count>1:
        return_value=count(return_value)[0]
        return_count=count(return_value)[1]
    if return_count<=1:
        if return_value==1:
            print("magic number")
        else:
            print("not magic number")
