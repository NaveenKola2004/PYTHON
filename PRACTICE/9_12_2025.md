def max_sub(arr):
    max_sum=arr[0]
    current_sum=0
    for n in arr:
        current_sum+=n
        if current_sum>max_sum:
            max_sum=current_sum
        
        if current_sum<0:
            current_sum=0
    return max_sum
print(max_sub([-2, 1, -3, 4, -1, 2, 1, -5, 4]))
