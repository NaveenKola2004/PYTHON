```python
def Trapping_Water(Heights):
    n=len(Heights)

    left_max=[0]*n
    right_max=[0]*n
    water=[0]*n

    # left max (prefix sum)

    left_max[0]=Heights[0]
    for i in range(1,n):
        left_max[i]=max(left_max[i-1],Heights[i])
    
    # right max (suffix sum)

    right_max[n-1]=Heights[n-1]
    for i in range(n-2,-1,-1):
        right_max[i]=max(right_max[i+1],Heights[i])
    
    total=0

    for i in range(n):
        water[i]=min(left_max[i],right_max[i])-Heights[i]
        total+=water[i]
    return total
n=int(input("Enter the number  : "))
l=[]
for _ in range(n):
    l.append(int(input()))
print(Trapping_Water(l))
```
