    count=0
    while n>0 :
        count+=1
        n//=10
    return count

    len=Count(n)
    l=[1]
    d=[]
    sample=1
    for i in range(len-1):
        sample=sample*2
        l.append(sample)
    while n>0:
        digits=n%10
        if digits==0:
            l.pop(0)
        else:

    return sum

A Leader is an element that is greater than all elements to its right.

Example:

[16, 17, 4, 3, 5, 2]

def find(n):
    Leaders=[]
    for i in range(len(n)):
        for j in range(i+1,len(n)):
            if n[i]<n[j]:
                break
        else:
            Leaders.append(n[i])
    print(Leaders)

def Anagram(w1,w2):
    w1_freq={}
    w2_freq={}
    for i in w1:
        if i in w1_freq:
            w1_freq[i]+=1
        else:
            w1_freq[i]=1
    for i in w2:
        if i in w2_freq:
            w2_freq[i]+=1
        else:
            w2_freq[i]=1
    if w1_freq==w2_freq:
        return "Anagram"
    else:
