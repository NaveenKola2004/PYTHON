def Fact(n):
    if n==0 or n==1:
        return 1
    return n* Fact(n-1)
n=5
for i in range(n):
    print(" " * (n - i), end="")
    for k in range(i+1):
        # ncr formula
        # n=i
        # r=k
        value= Fact(i) // (Fact(k) * Fact(i-k))
        print(value,end=" ")
